    <script>
        let animationTimeout;
        
        function generateLink() {
            const query = document.getElementById('searchInput').value.trim();
            if (!query) {
                alert('Please enter a search query!');
                return;
            }
            
            const baseUrl = window.location.origin + window.location.pathname;
            const encodedQuery = encodeURIComponent(query);
            const fullUrl = `${baseUrl}?q=${encodedQuery}`;
            
            document.getElementById('shareLink').textContent = fullUrl;
            document.getElementById('shareSection').style.display = 'block';
            
            // Smooth scroll to share section
            document.getElementById('shareSection').scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }
        
        function copyLink() {
            const linkText = document.getElementById('shareLink').textContent;
            navigator.clipboard.writeText(linkText).then(() => {
                const originalText = document.getElementById('shareLink').textContent;
                document.getElementById('shareLink').textContent = '✓ Copied to clipboard!';
                document.getElementById('shareLink').style.color = '#FFD700';
                setTimeout(() => {
                    document.getElementById('shareLink').textContent = originalText;
                    document.getElementById('shareLink').style.color = '#ffffff';
                }, 2000);
            }).catch(() => {
                // Fallback for older browsers
                const textArea = document.createElement('textarea');
                textArea.value = linkText;
                document.body.appendChild(textArea);
                textArea.select();
                document.execCommand('copy');
                document.body.removeChild(textArea);
                
                const originalText = document.getElementById('shareLink').textContent;
                document.getElementById('shareLink').textContent = '✓ Copied!';
                document.getElementById('shareLink').style.color = '#FFD700';
                setTimeout(() => {
                    document.getElementById('shareLink').textContent = originalText;
                    document.getElementById('shareLink').style.color = '#ffffff';
                }, 2000);
            });
        }
        
        function shareOnTwitter() {
            const linkText = document.getElementById('shareLink').textContent;
            const tweetText = encodeURIComponent('Here, let me search that for you: ');
            window.open(`https://twitter.com/intent/tweet?text=${tweetText}&url=${encodeURIComponent(linkText)}`, '_blank');
        }
        
        function shortenLink() {
            alert('Link shortening feature coming soon!');
        }
        
        function previewAnimation() {
            const query = document.getElementById('searchInput').value.trim();
            if (!query) {
                alert('Please enter a search query!');
                return;
            }
            
            // Clear any existing animation
            clearTimeout(animationTimeout);
            
            // Reset preview
            document.getElementById('previewSection').classList.add('active');
            document.getElementById('typingText').textContent = '';
            document.getElementById('fakeResults').classList.remove('show');
            document.getElementById('cursor').style.display = 'inline-block';
            
            // Smooth scroll to preview
            document.getElementById('previewSection').scrollIntoView({ behavior: 'smooth', block: 'nearest' });
            
            // Start typing animation
            typeText(query, 0);
        }
        
        function typeText(text, index) {
            if (index < text.length) {
                document.getElementById('typingText').textContent += text[index];
                animationTimeout = setTimeout(() => typeText(text, index + 1), 80);
            } else {
                // Show results after typing completes
                setTimeout(() => {
                    document.getElementById('cursor').style.display = 'none';
                    document.getElementById('fakeResults').classList.add('show');
                    
                    // Simulate redirect after 2 seconds
                    setTimeout(() => {
                        const perplexityUrl = `https://www.perplexity.ai/search?q=${encodeURIComponent(text)}`;
                        document.getElementById('fakeResults').innerHTML = `
                            <p style="color: #FFD700;">✓ Ready!</p>
                            <p style="margin-top: 10px;">Redirecting to Perplexity.ai...</p>
                            <p style="margin-top: 15px; font-size: 0.85rem;">
                                <a href="${perplexityUrl}" target="_blank" style="color: #FFD700; text-decoration: none;">Click here if not redirected automatically</a>
                            </p>
                        `;
                    }, 2000);
                }, 500);
            }
        }
        
        // Check for query parameter on page load
        window.onload = function() {
            const urlParams = new URLSearchParams(window.location.search);
            const query = urlParams.get('q');
            
            if (query) {
                document.getElementById('searchInput').value = query;
                // Hide the main interface
                document.querySelector('.tagline').style.display = 'none';
                document.querySelector('.search-container').style.display = 'none';
                document.querySelector('.button-group').style.display = 'none';
                document.querySelector('.social-links').style.display = 'none';
                
                // Show preview section immediately
                document.getElementById('previewSection').style.display = 'block';
                document.getElementById('previewSection').classList.add('active');
                
                // Start animation after a brief delay
                setTimeout(() => {
                    typeText(query, 0);
                    
                    // Actually redirect after animation completes
                    setTimeout(() => {
                        window.location.href = `https://www.perplexity.ai/search?q=${encodeURIComponent(query)}`;
                    }, 5000);
                }, 500);
            }
        };
        
        // Add enter key support
        document.getElementById('searchInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                generateLink();
            }
        });
    </script>
